```text
  rsp 0 8 4 
  rsp 1 12 4 
  rsp 2 2 2 
  rsp 3 2 2 
  rsp 4 5 16 
  rsp 5 8 16

  rsp 7 15 16 
  endrespawn

```

```text
respawn {
  rsp [id] [x (from left)] [y (from bottom)]
  ...
  endrespawn
}
```
